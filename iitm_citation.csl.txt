        <link href="http://www.example.com/custom-citation-style" rel="self"/>
        <category citation-format="author-date"/>
        <summary>A custom citation style based on the reference provided.</summary>
    </info>
    
    <macro name="author">
        <names variable="author">
            <name and="text" delimiter=", " initialize-with="."/>
        </names>
    </macro>
    
    <macro name="year">
        <date variable="issued">
            <date-part name="year" prefix=" (" suffix=")"/>
        </date>
    </macro>
    
    <macro name="title">
        <text variable="title" suffix="."/>
    </macro>
    
    <macro name="journal">
        <group delimiter=" ">
            <text variable="container-title" font-style="italic"/>
            <text variable="volume" font-weight="bold"/>
            <text variable="page" prefix=", "/>
        </group>
    </macro>
    
    <citation>
        <layout delimiter=" ">
            <text macro="author"/>
            <text macro="year"/>
        </layout>
    </citation>
    
    <bibliography>
        <layout delimiter=" ">
            <text macro="author"/>
            <text macro="year"/>
            <text macro="title"/>
            <text macro="journal" suffix="."/>
        </layout>
    </bibliography>
